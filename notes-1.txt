Локализация:

Локализация осуществляется созданием и правкой файла в директории с расширением. Он может находится в корне директории с расширением или в поддиректории dictionaries.
например для расширения sample-extension:
файл с именем ru.dict.sample-extension.php

<?php
/**
 * Localized data
 *
 * @copyright   Copyright (C) 2013 XXXXX
 * @license     http://opensource.org/licenses/AGPL-3.0
 */
 
Dict::Add('RU RU', 'Russian', 'Русский', array(
	// Dictionary entries go here\
	'Class:Monitor' => 'Монитор',
	'Class:Monitor+' => 'Устройство вывода изображения',
	'Class:Monitor/Attribute:technology' => 'Технология',
	'Class:Monitor/Attribute:technology+' => 'Технология для вывода изображения',
	'Class:Monitor/Attribute:technology/Value:crt' => 'ЭЛТ',
	'Class:Monitor/Attribute:technology/Value:lcd' => 'ЖКД',
));
?>
